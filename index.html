const express = require('express'); const { createProxyMiddleware } = require('http-proxy-middleware'); const app = express(); const PORT = 3000; // صفحه اصلی پروکسی app.get('/', (req, res) => { res.send(` Web Proxy

Web Proxy Service

`); }); // میدلور پروکسی app.use('/proxy', createProxyMiddleware({ target: 'http://example.com', // مقدار پیش‌فرض changeOrigin: true, pathRewrite: (path, req) => { const targetUrl = req.query.url; if(targetUrl) { return new URL(targetUrl).pathname; } return path; }, router: (req) => { return req.query.url ? new URL(req.query.url).origin : null; }, onProxyRes: (proxyRes) => { proxyRes.headers['access-control-allow-origin'] = '*'; } })); app.listen(PORT, () => { console.log(`Proxy server running on http://localhost:${PORT}`); });
